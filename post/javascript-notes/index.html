1. Async and Defer keyword

We can call external javascript by

1

<script src="operation.js" async></script>

Above code tells the browser to continue downloading the HTML content once the script tag element has been reached. Async scripts will download the script without blocking rendering the page and will execute it as soon as the script finishes downloading. You get no guarantee that scripts will run in any specific order, only that they will not stop the rest of the page from displaying. It is best to use async when the scripts in the page run independently from each other and depend on no other script on the page.

1
2
3
4
5

<script async src="js/vendor/jquery.js"></script>

<script async src="js/script2.js"></script>

<script async src="js/script3.js"></script>

If script2.js depends jquery.js to work then this code may not run properly because loading script2.js may finish before loading jquery.js. Defer will run the scripts in the order they appear in the page and execute them as soon as the script and content are downloaded:

1
2
3
4
5

<script defer src="js/vendor/jquery.js"></script>

<script defer src="js/script2.js"></script>

<script defer src="js/script3.js"></script>

All the scripts with the defer attribute will load in the order they appear on the page. So in the above example, we can be sure that jquery.js will load before script2.js and script3.js and that script2.js will load before script3.js.

//block.js
function log(num) {
    console.log(num);
}
function id(idValue) {
    return document.getElementById(idValue);
}


window.onload = function () {

    let weather = id("weather");
    let para = id("par");

    weather.addEventListener("change", chooseWeather);

    function chooseWeather() {
        let choice = Number(weather.value);

        let result = "";

        switch (choice) {
            case 1:
                result = "Sunny selected";
                break;

            case 2:
                result = "Rainy selected";
                break;

            case 3:
                result = "Snowing selected";
                break;

            default:
                result = "No choice";
        }

        para.textContent = result;
    }

}

    return document.getElementById(idValue);
}

window.onload = function () {

    let btnShow = id("btnShow");
    let btnRemove = id("btnRemove");

    function log1() {
        log("log1 execute");
    }

    function log2() {
        log("log2 execute");
    }

    btnShow.addEventListener("click", log1);
    btnShow.addEventListener("click", log2);

    btnRemove.addEventListener("click", function () {
        btnShow.removeEventListener("click", log1);
    });
}

After clicking btnRemove, btnShow click event only executes log2 function.

function random(number) {
    return Math.floor(Math.random() * (number + 1));
}

function bgChange(event) {
    var rndCol = 'rgb(' + random(255) + ',' + random(255) + ',' + random(255) + ')';    
    
    event.target.style.backgroundColor = rndCol;

    log(event);
    log(event.target);
}

window.onload = function () {

    let btnShow = id("btnShow");    

    btnShow.addEventListener("click", bgChange);
}

event.target object in bgChange method refers btnShow button element. We can observe the content of event in browser console tab.

function log(num) {
    console.log(num);

function id(idValue) {
    return document.getElementById(idValue);
}

window.onload = function () {

    let form = id("form");
    let para = id("para");
    let fname = id("fname");
    let lname = id("lname");

    form.addEventListener("submit", function (e) {

        e.preventDefault();

        if (fname.value === '' || lname.value === '') {
            para.textContent = "You need to fill in both names !";
        } else {
            para.textContent = "Form submit is successful";
            log({ firstName: fname.value, lastName: lname.value });
        }
    });
}
