using App.Models;
using Microsoft.Extensions.Configuration;

    public class HomeController : Controller
    {
        private readonly IConfiguration _config;

        public HomeController(IConfiguration config)
        {
            _config = config;
        }

        public string Index()
        {
            var model = new ConfigModel();
            _config.GetSection("custom").Bind(model);

            return model.Name + " | "+ 
                   model.Surname + " | " +
                   model.Number;
        }

You should see

1

output.

Tip

We can override appsettings.json file settings by changing appsettings.(Development | Production).json file settings.

@using Microsoft.Extensions.Configuration
@using App.Models;
@inject IConfiguration Configuration

@{
    ViewData["Title"] = "About Page";
    ConfigModel configModel=new ConfigModel();
    Configuration.GetSection("custom").Bind(configModel);
}

  <dl class="dl-horizontal">
      <dt>Name</dt>
      <dd>@configModel.Name</dd>

      <dt>Surname</dt>
      <dd>@configModel.Surname</dd>

      <dt>Number</dt>
      <dd>@configModel.Number</dd>
  </dl>
