#Adding project to solution
dotnet sln add App/App.csproj

#Removing a project from solution
dotnet sln remove App/App.csproj

#Create .net core classlib
dotnet new classlib -o Domain -f netcoreapp2.2

#Adding migrations
dotnet ef migrations add Student_Table_Added

#Apply migrations to database
dotnet ef database update

#Deleting database
dotnet ef database drop

#Initialize a website in your current folder
hexo init

#Create a new draft article with name draft1
hexo new draft "draft1"

#Publishes a draft article to post
hexo publish draft "draft1"

#Create a new post with name "Hexo-Start"
hexo new "Hexo-Start"  or 
hexo new post "Hexo-Start"

#Starts a local server. By default, this is at http:##localhost:4000/
hexo server

#Generates static files then deploying blog
hexo deploy -g

#Cleans the cache file (db.json) and generated files (public)
hexo clean

#List all routes
hexo list [post,page,tag or category]
